\frame{\frametitle{What makes a good presentation?}

\begin{itemize}
\item Concise slides - no paragraphs, rarely sentences, mostly fragments and phrases
\item A calm, prepared presenter
\item Clear visuals
\item Organization
\begin{itemize}
\item Color
\item Font
\item Layout
\end{itemize}
\end{itemize}
}

\frame{\frametitle{What makes a not-so-good presentation?}

\begin{itemize}
\item Wordy, complicated slides
\item Table of contents or outline
\end{itemize}
